var map = initMap({ tilt: 10, center: [121.422,30.993], zoom: 12, }); function initMap(options) { options = Object.assign({ tilt: 60, heading: 0 }, options); var map = new BMapGL.Map('map_container'); map.enableKeyboard(); map.enableScrollWheelZoom(); map.enableInertialDragging(); map.enableContinuousZoom(); if (options.center && options.zoom) { map.centerAndZoom(new BMapGL.Point(options.center[0], options.center[1]), options.zoom); } map.setTilt(options.tilt); map.setHeading(options.heading); return map; } var bd = new BMapGL.Boundary(); function drawregion(region){ bd.get(region, function (rs) { var hole = new BMapGL.Polygon(rs.boundaries, { fillColor: '', fillOpacity: 0.1, strokeStyle: 'dashed' }); map.addOverlay(hole); }); } drawregion('闵行区 马桥');
